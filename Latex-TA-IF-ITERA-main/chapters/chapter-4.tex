\section{Hasil Penelitian} \label{IV.Hasil}

\subsection{Pra-pemrosesan dan Eksplorasi Data}
\begin{itemize}
    \item \textbf{Sumber Data:} Menggunakan dataset sekunder SSGI 2024 yang mencakup 38 provinsi.
    \item \textbf{Kelengkapan Data:} Data agregat resmi terverifikasi 100\% terisi, tidak terdapat \textit{missing value}, sehingga tahap imputasi tidak diperlukan.
    \item \textbf{Seleksi Fitur:} Mengeliminasi variabel non-determinan agar model fokus pada faktor utama berdasarkan teori kesehatan masyarakat.
\end{itemize}

\subsection{Implementasi \textit{Random Forest} dan \textit{SHAP}}
\begin{itemize}
    \item \textbf{Pemilihan \textit{Random Forest}:} Algoritma ini diimplementasikan karena ketahanannya terhadap \textit{overfitting} pada dataset berukuran kecil (38 observasi).
    \item \textbf{Integrasi \textit{SHAP}:} Digunakan tidak hanya untuk mengukur tingkat kepentingan variabel, tetapi juga memetakan arah dampaknya terhadap angka \textit{stunting}.
\end{itemize}

\subsection{Skenario \textit{Hyperparameter Tuning}}
\begin{itemize}

\section{Analisis Hasil Penelitian} \label{IV.Analisis}

\subsection{Interpretasi Global (\textit{Global Explanation})}
\begin{itemize}
    \item \textbf{Peringkat Variabel:} Analisis fitur paling determinan secara nasional menggunakan \textit{SHAP Bar Plot}.
    \item \textbf{Arah Kontribusi:} Analisis \textit{SHAP Summary Plot} untuk melihat bagaimana kondisi variabel yang memburuk (titik merah) mendorong kenaikan angka \textit{stunting}.
\end{itemize}

\subsection{Interpretasi Lokal (\textit{Local Explanation})}
\begin{itemize}
    \item \textbf{Bedah Provinsi:} Analisis spesifik pada provinsi dengan prevalensi ekstrem menggunakan \textit{SHAP Waterfall Plot}.
    \item \textbf{Mekanisme Prediksi:} Menjelaskan pergeseran dari nilai rata-rata nasional (\textit{Base Value}) menuju nilai prediksi akhir akibat dorongan faktor lokal (balok merah/biru).
\end{itemize}

\subsection{Pembahasan dan Rekomendasi Kebijakan}
\begin{itemize}
    \item \textbf{Kesesuaian Teori:} Konfirmasi temuan \textit{multiplier effect} antarvariabel \textit{stunting} dengan pedoman gizi Kementerian Kesehatan.
    \item \textbf{Rekomendasi Spesifik:} Usulan intervensi kebijakan gizi yang tidak dipukul rata, melainkan disesuaikan dengan akar masalah terbesar di masing-masing provinsi berdasarkan \textit{Waterfall Plot}.
\end{itemize}
